$(document).ready(function(){
	// first make sure they have cookies enabled, so we don't annoy the heck out of people
	
	// set a test cookie with short expires
	var date = new Date();
	var minutes = 2;
	date.setTime(date.getTime() + (minutes * 60 * 1000));
	
	$.cookie('cookie_check', '1', { path: '/', expires: date });
	if ($.cookie('cookie_check') == '1') {
		// cookies are enabled, now do our thing
	
		var date = new Date();
		var now = date.getTime();
		
		// check if we should be showing anything
		var showOverlay = false;
		if ($.cookie('show_2015_bracket_pack_after')) {
			var cookieValue = $.cookie('show_2015_bracket_pack_after');
			if (cookieValue < now) {
				showOverlay = true;
			}
			//alert(cookieValue);
			//var difference = cookieValue - now;
			//alert(difference);
		} else {
			//alert("No cookie!");
			showOverlay = true;
		}
		
		// big layout problems in IE7, punting for now and just not showing the popup if it's IE7 or lower
		if ( $.browser.msie ) {
			if (parseInt($.browser.version, 10) <= 7 ) {
				showOverlay = false;
			}
		}
		
		if (showOverlay == true) {
			var mm13timer = setTimeout(openOverlay, 4000);
		}
	
	$("#mm13overlay #remind-me, #cboxOverlay").click(function(e) {
		e.preventDefault();
		// make sure they didn't subscribe already, in which case we don't want to show them this again:
		var currCookieValue = $.cookie('show_2015_bracket_pack_after');
		if (!currCookieValue || ((currCookieValue - now) < 24 * 60 * 60 *1000)) {
			var date = new Date();
			var hours = 18; // how long to wait before showing the popup again
			var cookieTime = date.getTime() + (hours * 60 * 60 * 1000);
			date.setTime(cookieTime);
			$.cookie('show_2015_bracket_pack_after', cookieTime, {
				path: '/',
				expires: date
			});
		}
		$.colorbox.close();
	});
	
	$("#mm13overlay #dont-ask, #mm13overlay .tr-close-button a").click(function(e) {
		e.preventDefault();
		var date = new Date();
		var days = 90;
		var cookieTime = date.getTime() + (days * 24 * 60 * 60 * 1000);
		date.setTime(cookieTime);
		$.cookie('show_2015_bracket_pack_after', cookieTime, {
			path: '/',
			expires: date
		});
		$.colorbox.close();
	});
	
	$("#mm13overlay #close-window").live("click", function(e) {
		e.preventDefault();
		$.colorbox.close();
	});
	
	$("#mm13overlay #subscribe-button").click(function(e) {
		e.preventDefault();
		$("#mm13overlay #subscribe-form").submit();
	});
	
	$("#mm13overlay #subscribe-form").submit(function(e) {
		e.preventDefault();
		var buttonText = $("#mm13overlay #subscribe-button").text();
		$("#mm13overlay #subscribe-button").removeClass('btn-green').addClass('btn-processing').text("Processing...");
		$.ajax({
			url: $("#mm13overlay #subscribe-form").attr('action'),
			data: 'ajax=true&email=' + $('#mm13overlay #email').val(),
			success: function(msg) {
				if (msg.slice(0,18) == "Success-subscribed") {
					var displayMsg = msg.slice(18);
				} else {
					var displayMsg = msg;
				}
				$('#mm13overlay #form-response p').html(displayMsg);
				$("#mm13overlay #subscribe-button").removeClass('btn-processing').addClass('btn-green');
				$("#mm13overlay .form-row span.error").removeClass('show');
				$("#mm13overlay .email").removeClass('error');
				$("#mm13overlay #form-response").removeClass('error success');
				
				// Success:
				if (msg.slice(0,7) == "Success") {
					$("#mm13overlay #form-response, #mm13overlay .action-links, #mm13overlay #remind-me").addClass("success");
					$("#mm13overlay #subscribe-button").hide();
					$("#mm13overlay #email").attr('disabled', 'disabled');
					$("#mm13overlay #dont-ask").remove();
					$("#mm13overlay .action-links").html('&nbsp;');
					var date = new Date();
					var days = 90;
					var cookieTime = date.getTime() + (days * 24 * 60 * 60 * 1000);
					date.setTime(cookieTime);
					$.cookie('show_2015_bracket_pack_after', cookieTime, {
						path: '/',
						expires: date
					});
				} else if (msg.slice(0,5) == "Error") { // Error
					$("#mm13overlay #form-response, #mm13overlay .email").addClass("error");
					$("#mm13overlay .form-row span.error").addClass('show');
					$("#mm13overlay #subscribe-button").text(buttonText);
				}
			}
		});

	});
	
	} // cookie check

	function openOverlay() {

				$("#cboxWrapper").height($("#cboxWrapper").height()-28); // only way to remove extra 28px black space at bottom
			}
		})
	}
